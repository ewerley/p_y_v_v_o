    """A Python class for the PSO optmizer

    Attributes
    ----------
    c1 : `float`, by default 1.
        Learning (cognitive) parameter
    c2 : `float`, by default 1.5
        Global learning (social) parameter
    wmin : `float`, by default 0.3
        Minimum weight
    wmax : `float`, by default 0.9
        Maximum weight 

    **General attributes**

    npart : `int`
        Number of particles
    dim : `int`
        Problem's dimension
    x : `numpy.ndarray`, (npart, dim)
        Position vector of the population
    v : `numpy.ndarray`, (npart, dim)
        Velocity vector of the population
    w : `float`
        Velocity weight
    xbest : `numpy.ndarray`, (npart, dim)
        Best position vector of each particle
    xbest_fitness : `numpy.ndarray`, (npart, 1) 
        Fitness value of xbest        
    gbest : `numpy.ndarray`, (npart, 1)
        Best position vector among all particles (global best)
    gbest_fitness : `float` 
        Fitness value of the global best
    gbest_fitness : `numpy.ndarray`, (itermax, 1)
        Fitness value of the global best at each iteration   
    """    

    def __init__(self, c1=1., c2=1.5, wmin=0.3, wmax=0.9):
   
        self.c1 = c1 # Learning factor
        self.c2 = c2 # Global Learning factor        
        self.wmin = wmin # Minimum weight
        self.wmax = wmax # Maximum weight

        # Methods' attributes
        self.npart = None
        self.dim = None    
        self.xmin, self.xmax = [], []
        self.iter = 0
        self.x = None # numpy.ndarray (npart, dim)
        self.v = None # numpy.ndarray (npart, dim)
        self.w = None # `float`
        
        self.gbest_fitness = None # `float``
        self.gbest_fitness_per_iteration = []
        self.gbest = None # numpy.ndarray (dim, 1) 
        self.xbest = None # numpy.ndarray (npart, dim)                
        self.xbest_fitness = None # numpy.ndarray (npart, 1)


[docs]    def optimize(self, fitnessfunc, npart, xmin, xmax, xstep=None, direction='min', 
                 itermax=200, discretization=None, xclamp=False, vclamp=False, 
                 vmin=None, vmax=None, show_progress=False, report=False):
        """Run the PSO optimizer

        Parameters
        ----------
        fitnessfunc : callable
            A callback function that takes a `numpy.ndarray` as argument and returns
            the corresponding fitness value.
        npart : `int`
            Number of particles
        itermax : int, by default 200
            Maximum number of iterations
        xmin : `numpy.ndarray`, (1, dim)
            Lower limit of each variable of the position vector
        xmax : `numpy.ndarray`, (1, dim)
            Upper limit of each variable of the position vector
        vmin : `numpy.ndarray`, (1, dim), optional
            Lower limit of each dimension of the particle's velocity
        vmax : `numpy.ndarray`, (1, dim), optional
            Upper limit of each dimension of the particle's velocity
        xstep : `numpy.ndarray`, (1, dim), optional
            Step size of each (discrete) variable of the position vector. A `None` 
            value indicates a continuous variable.
        direction : {'min', 'max'}, by default 'min'
            Optimization direction. If 'min', seeks for a minimum.
            If 'max', seeks for a maximum. 
        discretization : {'iter', 'mid-iter ,  'end'}, by default None
            Discretization strategy. In the 'iter' case, the discrete variables
            are rounded to the next discrete value at every iteration. In the 
            'mid-iter' case, the discretization is enabled only after ``maxiter``/2 
            iterations. In the 'end' strategy, the discretization is performed only 
            after the final convergence.
        xclamp : bool, by default False
            If `True`, each element of the position vector is forced to remain between 
            the limits in ``xmin`` and ``xmax``.
        vclamp : bool, by default False
            If `True`, each element of the velocity vector is forced to remain between 
            the limits in ``vmin`` and ``vmax``.
        show_progress : bool, optional
            _description_, by default False
        """        

        self.fitnessfunc = fitnessfunc
        self.npart = npart
        self.dim = dim = len(xmin)
        self.xmin = xmin = np.array(xmin)
        self.xmax = xmax = np.array(xmax)
        self.xstep = xstep
        self.discretization = discretization
        self.direction = direction
        self.itermax = itermax 
        self.xclamp = xclamp
        self.xbest = np.zeros((self.npart, self.dim))
        self.xbest_fitness = np.zeros(self.npart)
        self.gbest_fitness_per_iteration = np.zeros(itermax)

        # Consistency check
        if (vclamp and (not vmin or not vmax)):
            print('Pso error: vclamp need values for vmin and vmax')
            exit(1)

        # Initialize particle, velocities, pbest and gbest
        self.x = np.zeros((self.npart, self.dim))        
        self.v = np.zeros((self.npart, self.dim))
        for p in range(npart):            
            self.x[p, :] = xmin + rand(dim) * (xmax - xmin)
            if discretization == 'iter':
                self.x[p, :] = self._round_to_discrete(self.x[p, :], xstep, xmin)
            self.v[p, :] = 0 if not vclamp else (vmin + rand(dim) * (vmax - vmin))
            self.xbest[p] = self.x[p]
            self.xbest_fitness[p] = fitnessfunc(self.x[p])

        self.gbest_fitness, self.gbest_pos = self._getBest(self.xbest_fitness)
        self.gbest = self.xbest[self.gbest_pos]

        x, v = self.x, self.v
        c1, c2 = self.c1, self.c2
        w, wmin, wmax = self.w, self.wmin, self.wmax
        npart, dim = self.npart, self.dim
        iter = self.iter

        if report or show_progress:
            print('* PSO results *\n')

        if report:
            print('Number of particles:', self.npart)
            print('Max. iterations:', self.itermax)
            print('Learning coefficients (c1,c2): ' + str(self.c1) + ', ' + str(self.c2))
            print('Weights (wmin, wmax): ' + str(self.wmin) + ', ' + str(self.wmax))
            print()

        if show_progress:
            from progress.bar import Bar
            bar = Bar('Optimizing', max=itermax, suffix='%(index)d/%(max)d (iterations)')            

        # Main loop
        while iter < itermax:           
            w = wmax - (wmax-wmin)*iter/itermax
            self.gbest_fitness_per_iteration[iter] = self.gbest_fitness

            for p in range(npart):
                # Update velocity
                v[p, :] = (w * v[p, :]
                           + c1*rand()*(self.xbest[p,:] - x[p,:])
                           + c2*rand()*(self.gbest - x[p,:]))

                # Velocity clamp
                if (vclamp):
                     v[p, :] = self._bound(v[p, :], vmax, vmin)

                # Update the position of the particles
                x[p, :] = x[p, :] + v[p, :]
                if (xclamp):
                     x[p, :] = self._bound(x[p, :], xmax, xmin)

                if discretization:
                    if discretization == 'iter' or \
                       (discretization == 'mid_iter' and iter > itermax/2):
                        x[p, :] = self._round_to_discrete(x[p, :], xstep, xmin)

                # Calculate fitness
                xfitp = fitnessfunc(self.x[p, :])
                # Check if pbest and gbest need to be updated
                if self._isbetterThan(xfitp, self.xbest_fitness[p]):
                    self.xbest[p, :] = x[p, :]
                    self.xbest_fitness[p] = xfitp 
                    if self._isbetterThan(xfitp, self.gbest_fitness):
                        self.gbest = x[p, :]
                        self.gbest_fitness = xfitp

            # Velocity calculation and position update
            iter += 1
            if show_progress:
                bar.next()
        
        if show_progress:
            bar.finish()
            print('')

        if discretization == 'end':
            self.gbest = self._round_to_discrete(self.gbest, xstep, xmin)
            self.gbest_fitness = fitnessfunc(self.gbest)


        
    def _getBest(self, xbests):
        """Return the value and position of the best element in the array ``xbests``

        If the optmizing direction is ``"min"``, the minimum element is returned.
        Otherwise, the maximum element is returned.

        Parameters
        ----------
        xbests : `numpy.ndarray` , (npart, 1)
            Input array

        Returns
        -------
        (best_value, best_pos) : `tuple`
            Best element's value and position
        """        

        if self.direction.lower() == 'min':
            best_value, best_pos = xbests.min(), xbests.argmin()
        elif self.direction.lower() == 'max':
            best_value, best_pos = xbests.max(), xbests.argmax()

        return best_value, best_pos

    def _isbetterThan(self, ga, gb):
        """Test if the real variable `ga` is better than `gb` in terms of fitness

        If the optmizing direction is ``"min"``, the results of ga < gb
        is returned. Otherwise, the result of ga > gb is returned.

        Parameters
        ----------
        ga : `float`
            Left-hand value
        gb : `float`
            Right-hand value

        Returns
        -------
        result : `bool`
            `True`, if ``ga`` is better than ``gb``.
        """    
        if self.direction.lower() == 'min':
            result = ga < gb
        elif self.direction.lower() == 'max':
            result = ga > gb
        return result

    def _bound(self, x, xmax, xmin):
        """Apply boundary conditions

        Each element of x is tested against the corresponding maximum and minimum
        limits. If there is a violation, the closest limit is enforced.
        """

        for i in range(len(x)):
            if x[i] > xmax[i]:
                x[i] = xmax[i]
            elif x[i] < xmin[i]:
                x[i] = xmin[i]
        return x

    def _round_to_discrete(self, x, xstep, xmin):     
        """Round to the nearest discrete variable

        Each element of x is rounded to its nearest discrete variable, based on the
        corresponding minimum values and step size.

        Parameters
        ----------
        x : `numpy.ndarray` , (1, ndim)
            Input array
        xstep : `numpy.ndarray` , (1, ndim)
            Step size for each element in ``x``
        xmin : `numpy.ndarray` , (1, ndim)
            Minimum value of each element of ``x``

        Returns
        -------   
        x_rounded : `numpy.ndarray` , (1, ndim)
            Array with rounded values

        Note
        ----
        Only the elements with ``xstep`` different than `None` are rounded.
        """        
        x_rounded = [
            xmin[c] + xstep[c]*((x[c]-xmin[c])/xstep[c]).round()
            if xstep[c] else x[c] for c in range(len(x))
            ]
        return x_rounded
