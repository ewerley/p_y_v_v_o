import matplotlib.pyplot as plt
from matplotlib.ticker import FormatStrFormatter


[docs]def fitness_plot(pso, xlabel = 'Iteration', ylabel = 'Fitness', subtitle= None):
    """Generate the fitness per iteration plot

    Parameters
    ----------
    pso : `pso.Pso`
        PSO object
    subtitle : `str`, optional
        Plot subtitle, by default None
    xlabel : `str`, optional
        Axis x label, by default 'Iteration'
    ylabel : `str`, optional
        Axis y label, by default 'Fitness'
    """        
    _, ax = plt.subplots()
    ax.grid()
    ax.set_xlabel(xlabel)
    ax.set_ylabel(ylabel)
    #ax.yaxis.set_major_formatter(FormatStrFormatter('%.1f'))
    ax.plot(pso.gbest_fitness_per_iteration)
    ax.set_title('PSO: {}\nnpart={}, c1={}, c2={}'.format(subtitle, pso.npart, pso.c1, pso.c2))    
    plt.show()
