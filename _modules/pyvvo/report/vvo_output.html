[docs]def info_report(vvo, options):
    """Print a short summary report

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    options : `list` [ `str` ]
        Types of reports to print
    """        
    if 'info' in options:
        print('\n pyVVO v' + __version__ )
        print('\n* System information *\n')
        print('System name:', vvo.grid.sysname)
        print('Number of buses:', vvo.grid.nbus)
        print('Number of nodes:', vvo.grid.nnodes)
        print('Number of branches:', vvo.grid.nbr)
        print()        
        if vvo.goal == 'min_losses':
            goal = 'loss minimization'
        elif vvo.goal == 'min_dv':
            goal = 'voltage-deviation minimization'
        print('Optimization goal:', goal)    
        print('Number of control variables:', vvo.nc)
        # print('Number of constraints:', vvo.nconstraints)
        print('Discretization strategy:', vvo.discretization)
        print()




[docs]def base_report(vvo, options):
    """Print the base case report

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    options : `list` [ `str` ]
        Types of reports to print
    """       
    if 'base' in options or 'base_controls' in options:
        # Base case information
        print('* Base-case report *\n')

        print('Controls\n')
        print_controls(vvo, u_opt=vvo.ubase)

        if vvo.goal == 'min_losses':
            print('Losses (kw): {:5g}'.format(vvo.ofbase/1000))
        elif vvo.goal == 'min_dv':

        print()

    if 'base' in options or 'base_voltages' in options:
        print('* Base-case voltages (pu) *\n')
        print_voltages(vvo)            

    if 'base_loads' in options:
        print('Load report\n')
        print_loads(vvo)

    if 'base_vchart' in options:
        print_voltage_chart(vvo, chart_title='{}: Base-case Line-to-neutral Voltages (p.u)')

    if 'base_flows' in options:
        print_flows(vvo)        




[docs]def optimal_report(vvo, options):
    """Print a report of the optimal results

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    options : `list` [ `str` ]
        Types of reports to print
    """        
    if 'opt' in options or 'opt_controls' in options:
        # Optimization results
        print('Optimal controls\n')
        
        print_controls(vvo, show_range=True)

        if vvo.goal == 'min_losses':
            print('Optimal Losses (kw): {:5g}'.format(vvo.of_opt/1000))
        elif vvo.goal == 'min_dv':
            print('Voltage deviation (pu): {:5g}'.format(vvo.of_opt))
            print('Losses (kw): {:5g}'.format(vvo._evalFitness(goal='min_losses')/1000))
        print()
 
        print('Penalty factors adopted\n')
        print(' {:<12}{}'.format('type', 'value'))
        print('-'*20)

        for p in vvo.penalties:
            if (p == 'kvar-cap' and not vvo.grid.capacitor) \
                    or (p == 'tap' and not vvo.grid.transformer) \
                    or (p == 'fp-inv' and not vvo.grid.generator) \
                    or (p == 'kva-inv' and not vvo.grid.generator):
                continue
            print(' {:<12}{}'.format(p, vvo.penalties[p]))
        print('-'*20)    

    if 'opt' in options or 'constraints' in options:
        print('\n* Constraints *\n')
        # Load Voltage
        print('Load constraints \n')
        print_loads(vvo)

    if 'opt' in options or 'opt_voltages' in options:
        print('* Voltages at the optimal solution (pu) *\n')
        print_voltages(vvo)  

    if 'opt' in options or 'opt_flows' in options:
        print_flows(vvo)

    if 'opt' in options or 'fitness_plot' in options:
        from pyvvo.optimizers.pso.report import fitness_plot
        print("Generating fitness plot..")
        print('\nClose the graph window to quit.\n')
        # Fitness plot
        if vvo.goal == 'min_losses':
            title = vvo.grid.sysname + ' loss minimization'
            vvo.opt.gbest_fitness_per_iteration /= 1E3 # printing in kw
            ylabel = 'Fitness [kW]'
        elif vvo.goal == 'min_dv':
            title = vvo.grid.sysname + ' voltage-deviation minimization'
            ylabel = 'Voltage [pu]'
        fitness_plot(vvo.opt, subtitle=title, ylabel=ylabel)

    if 'opt' in options or 'opt_vchart' in options:
        print_voltage_chart(vvo)




[docs]def print_controls(vvo, u_opt=None, show_range=False):
    """Print the control variables

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    u_opt : `list`, optional
        Array of optimal values
    show_range : bool, by default False
        If `False`, shows the minimum and maximum control ranges
    """    
    if not u_opt:
        u_opt = vvo.u_opt

    fmt = '{:^10s}{:^10s}{:^10s}'
    var = ['Device', 'variable', 'value']
    dashed_line = '-'*30
    if show_range:
        fmt += '{:^15s}'
        var += ['range']
        dashed_line += '-'*15

    print(fmt.format(*var))
    print(dashed_line)

    for u in range(vvo.nc):
        if vvo.uid[u][0] == 'generator':
            device = vvo.uid[u][1]
            variable = 'pf'
            kvar = u_opt[u]
            generator = vvo.grid.generator[vvo.uid[u][1]]
            kw = float(generator['kw'])
            value = np.cos(np.arctan2(kvar,kw))
            if show_range:        
                urange = '[{}, {}]'.format(generator['minpf'], generator['maxpf'])
        else:
            device = vvo.uid[u][1]
            variable = vvo.uid[u][2]
            value = u_opt[u]
            if show_range:
                urange = '[{}, {}]'.format(vvo.umin[u], vvo.umax[u])
        
        var = [device, variable, '{:^10g}'.format(value)]                
        if show_range:
            var += [urange]
        print(fmt.format(*var))

    print(dashed_line)
    print()

    """Print the loads available and the line-to-neutral voltages

    Parameters
    ----------

    loadbuses = [vvo.grid.load[id_]['bus1'].split('.')[0] for id_ in vvo.grid.load]
    loadkeys = list(vvo.grid.load.keys())
    max_loadbus_length = len(max(loadbuses, key=len))
    max_loadkey_length = len(max(list(vvo.grid.load.keys()), key=len))

    fmt = '{:' + str(max_loadkey_length + 3) + 's}' \
        + '{:^' + str(max_loadbus_length + 4) + 's}' \
        + '{:^7s}' + '{:<5}  {:^10s} {:^10s} {:^10s}   {}'

    fmth1 = '{:' + str(max_loadkey_length + 3) + 's}' \
        + '{:^' + str(max_loadbus_length + 4) + 's}' \
        + '{:^7s}' + '{:^5}  {:^30s}   {}'        

    print(fmth1.format('load', 'bus', 'conn',  'nodes', ' Line-to-neutral Voltages (pu)', '     range'))
    print(fmt.format('', '', '',  '', 'V @ ph1', 'V @ ph2', 'V @ ph3', '[Vmin,Vmax]'))
    print('-'*(max_loadkey_length + 3 + max_loadbus_length + 4 + 40 + 8 +12))
    
    loadbuses_fromroot_indexes = [vvo.grid.sweepfromroot_buses.index(l) for l in loadbuses]
    # loadbuses_sorted = [i[1] for i in sorted(list(zip(loadbuses_fromroot_indexes, loadbuses)))]
    loadkeys_sorted = [i[1] for i in sorted(list(zip(loadbuses_fromroot_indexes, loadkeys)))]

    # Sort according to the substation proximity:
    # loadbuses = sorted(set([vvo.grid.load[id_]['bus1'].split('.')[0] for id_ in vvo.grid.load]))
    # loadbuses_fromroot_indexes = [vvo.grid.sweepfromroot_buses.index(l) for l in loadbuses]
    # loadbuses = [i[1] for i in sorted(list(zip(loadbuses_fromroot_indexes, loadbuses)))]

    Vnodes_min = 1e5 * np.ones(3)
    Vnodes_max = np.zeros(3)
    for key in loadkeys_sorted:

        # for id_ in vvo.grid.load:
        #     # find first load connected in the bus
        #     if vvo.grid.load[id_]['bus1'].split('.')[0] == bus:
        #         break
            # objetivo: capturar ids e fases das cargas na mesma barra (depois reduzir as fases)
        bus, *load_nodes = vvo.grid.load[key]['bus1'].split('.')
        # phases = vvo.grid.load[key]['phases']
        vminpu = float(vvo.grid.load[key]['vminpu'])
        vmaxpu = float(vvo.grid.load[key]['vmaxpu'])
        bus_nodes = vvo.grid.busnodes[bus]['nodes']    
        if 'conn' not in vvo.grid.busnodes[bus].keys():
            conn = 'wye'
        else:
            conn = vvo.grid.busnodes[bus]['conn']
        inodes = vvo.grid.busnodes[bus]['inodes']
        V =  abs(vvo.grid.E[inodes])*3**0.5 / vvo.grid.vbase[inodes] # voltage magnitude in pu
        nodes_index = np.array(list(bus_nodes)).astype(int)-1
        load_nodes = ' ' + ''.join(load_nodes)

        V3ph_str = np.array(['-','-','-'], dtype='U32')
        urange = '[{}, {}]'.format(vminpu, vmaxpu)
        for n, node in enumerate(nodes_index):

                Vnodes_min[node] = V[n]

        print(fmt.format(key, bus, conn, load_nodes, *V3ph_str, urange))    

    print('-'*(max_loadkey_length + 3 + max_loadbus_length + 4 + 40 + 8 + 12))

        + '{:^' + str(max_loadbus_length + 4) + 's}' \
        + '{:^7s}' + '{:^5s} ' + '{:^10.5f} {:^10.5f} {:^10.5f}'
    print(fmt.format('', '', '', '(Vmax)', *Vnodes_max ))
    print(fmt.format('', '', '', '(Vmin)', *Vnodes_min ))
    print()




[docs]def print_flows(vvo):
    """Print a power flow (kW/kVar) report

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    """       
    print ('* Power flow report * \n')
    branch_list = []
    for branch in vvo.grid.branches:
        k = vvo.grid.branches[branch]['bus1']
        m = vvo.grid.branches[branch]['bus2']
        for e, n in enumerate(vvo.grid.branches[branch]['nodes2']):
            if e ==0:
                branch_list += [(k + '-' + m, n)]
            else:
                branch_list += [(' '*len(k + '-' + m ), n)]

    maxlenbr = max(len(max(branch_list)), 13)
    fmth = '{:^' + str(maxlenbr+1) + '} {:^3s} {:>10} {:>12}'
    fmtr = '{:^' + str(maxlenbr+1) + '} {:^3s} {:>10.4g} {:>12.4g}'    
    print(fmth.format('branch (k-m)','@node','Pkm (kW)','Qkm (kvar)'))
    dashed_line = '-'*(maxlenbr + 31)
    print(dashed_line)
    for i in range(len(vvo.grid.Skm)):
        skm = vvo.grid.Skm[i]/1000
        print(fmtr.format(*branch_list[i], skm.real, skm.imag))
    print(dashed_line)
    print()




[docs]def print_voltages(vvo):
    """Print line-to-neutral per-unit voltages

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    """      
    maxbus_length = len(max(vvo.grid.orderbus, key=len))
    fmt = '{:^' + str(maxbus_length + 4) + 's}{:^7s} {:^11s} {:^11s} {:^11s}'
    print(fmt.format('bus', 'kV base ',  '  V @ ph1 ', ' V @ ph2 ', ' V @ ph3 '))
    print('-'*(maxbus_length + 6 + 40))

    for bus in vvo.grid.sweepfromroot_buses:
        nodes = vvo.grid.busnodes[bus]['nodes']    
        inodes = vvo.grid.busnodes[bus]['inodes']
        V =  abs(vvo.grid.E[inodes])*3**0.5 / vvo.grid.vbase[inodes] # voltage magnitude in pu
        vbuses_base = vvo.grid.vbase[inodes]
        assert(sum(vbuses_base- vbuses_base[0]) == 0)
        vbus_base = vbuses_base[0]
        nodes_index = np.array(list(nodes)).astype(int)-1
        V3ph_str = np.array(['-','-','-'], dtype='U32')
        for n, node in enumerate(nodes_index):
            V3ph_str[node] = '{:^.5f}'.format(V[n])                            
        print(fmt.format(bus, ' {:^7g}'.format(vbus_base/1000), *V3ph_str))    

    print('-'*(maxbus_length + 6 + 40))
    print()    




[docs]def print_voltage_chart(vvo, chart_title=None):
    """Print the loads available and the line-to-neutral voltages

    Parameters
    ----------
    vvo : `pyvvo.Vvo`
        Vvo object
    chart_title : `str`, default None
        Chart title
    """            
    print ('Generating voltage chart... \n')

    if not chart_title:
        chart_title = 'System {}: Optimal Line-to-neutral Voltages (p.u)'

    loadbuses = set([vvo.grid.load[id_]['bus1'].split('.')[0] for id_ in vvo.grid.load])
    Vmin_load = dict.fromkeys(loadbuses)
    Vmax_load = dict.fromkeys(loadbuses)
    for bus in loadbuses:
        Vmin_load[bus] = []
        Vmax_load[bus] = []

    for key in vvo.grid.load:
        bus = vvo.grid.load[key]['bus1'].split('.')[0]
        vmin = float(vvo.grid.load[key]['vminpu'])
        vmax = float(vvo.grid.load[key]['vmaxpu'])
        Vmin_load[bus] += [vmin]
        Vmax_load[bus] += [vmax]

    # Test here if all loads in a same bus have the same limits
    for bus in loadbuses:       
        assert (sum(np.array(Vmin_load[bus])-Vmin_load[bus][0]) == 0.)
        assert (sum(np.array(Vmax_load[bus])-Vmax_load[bus][0]) == 0.)

    V = ([None]*vvo.grid.nbus, [None]*vvo.grid.nbus, [None]*vvo.grid.nbus)
    Vmin = dict.fromkeys(vvo.grid.sweepfromroot_buses)
    Vmax = dict.fromkeys(vvo.grid.sweepfromroot_buses)
    for k, v in Vmin_load.items():
        Vmin[k] = v[0]
    for k, v in Vmax_load.items():       
        Vmax[k] = v[0]

    for k, bus in enumerate(vvo.grid.sweepfromroot_buses):
        for p in vvo.grid.busnodes[bus]['nodes']:
            inode = vvo.grid.busnodes[bus][p]
            Vpu = abs(vvo.grid.E[inode])*3**0.5 / vvo.grid.vbase[inode]
            V[int(p)-1][k] = Vpu

    plt.rcParams['grid.color'] = (0.9, 0.9, 0.9, .1)
    _, ax = plt.subplots()
    ax.grid()
    ax.set_xlabel('bus')
    ax.set_ylabel('Vln (pu)')
    #ax.yaxis.set_major_formatter(FormatStrFormatter('%.1f'))
    for p in range(3):
        label = 'abc'
        color = 'rgb'
        ax.plot(vvo.grid.sweepfromroot_buses, V[p], label='V' + label[p], marker="v", color=color[p])

    ax.plot(vvo.grid.sweepfromroot_buses, Vmin.values(), linestyle='',
            label='Vmin', color='black', marker="_", markersize=20)        
    ax.plot(vvo.grid.sweepfromroot_buses, Vmax.values(), linestyle='', 
            label='Vmax', color='black', marker="_", markersize=20)        

    ax.set_title(chart_title.format(vvo.grid.sysname))    
    plt.legend(loc="lower left")
    plt.show()
